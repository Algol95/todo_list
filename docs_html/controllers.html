 21    def createTask(db: Session, title: str, description: str, state_id: int, username: str):
 22        """Crea una nueva tarea en BBDD
 23
 24        Args:  
 25            `db (Session):` Sesión activa de SQLAlchemy  
 26            `title (str):` Título de la tarea  
 27            `description (str):` Descripción de la tarea  
 28            `state_id (int):` ID del estado de la tarea  
 29            `username (str):` Nombre de usuario del dueño de la tarea  
 30
 31        Raises:

 34            Exception: El usuario no existe  
 35
 36        Returns:
 37            str: Mensaje de tarea creada o mensaje de error
 38        """
 39        try:
 40            if not title or not description:
 41                raise Exception("El título y la descripción no pueden estar vacíos")
 42            if not state_id:
 43                raise Exception("Se debe incluir una id de estado")
 44            user = UserController.findUserByUsername(db, username)

 48            db.add(task)
 49            db.commit()
 50            db.refresh(task)
 51            return f"\n\033[32mCreaste una tarea:\n {task}\033[0m"
 52        except Exception as e:
 53            db.rollback()
 54            return f"\033[31m\n❌ {e}\033[0m"
